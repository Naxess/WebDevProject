// EVENT LISTENERS
$('button').eq(0).click(function(){ // Adds an event listener to the element that performs an action when clicked
    console.log("HELLO!");
});

$('button').eq(1).click(function(){
    $(this).text("Clicked!"); // Using the 'this' keyword to select the current element
});

$('button').eq(1).dblclick(function(){ // performs an action when a double-click is performed on the element
    $(this).text("Bye!");
});

$('#wave_button').mouseover(function(){
    $(this).css("background-color","yellow"); // the css function allows changing the css properties via javascript
});

$('#wave_button').mouseout(function(){
    $(this).css("background-color","inherit"); // the 'inherit' value changes the color back to the default value
});

$('#hello_button').mouseover(function(){
    $(this).toggleClass('peepee_poopoo_class');
});

$('#bye_button').mouseover(function(){
    $(this).toggleClass('bingbing_bongbong_class');
});

$('#text_0').keypress(function(){

});

$('#text_1').on('keypress', function(){
