Arrays

    var letters = ['A', 'B', 'C']
    letters;                // Output: ['A', 'B', 'C']
    letters[1];             // Output: 'B'
    letters[1] = 'smol b';
    letters[1];             // Output: 'smol b'

    var mixed_array = ['A', true, 123, ['x', 'y', 'z'], [2, 4 ,6]]

    // Array Length
    letters.length;         // Output: 3
    [1,2,3,4].length;       // Output: 4

    // Concatenation
    var combined_arrays = ['a', 'b'].concat(['c', 'd']);
    combined_arrays;        // Output: ['a', 'b', 'c', 'd']

    // Adding new elements
    [1,2,3,4,5,6].push(7);  // Output: 7 (The push() method returns the length of the array.)
    ['a','b'].push('c');    // Output: 3

    // Removing elements. Pop removes the last element from the array
    var abc = ['a','b','c'];
    abc.pop();              // Output: 'c'

    // Reverses the order of the elements
    var reversed_arr = ['c','d','a','b'].reverse();
    reversed_arr;           // Output: ['b', 'a', 'd', 'c']

    // Sorts the elements in an array
    var sorted_arr = ['c','d','a','b'].sort();
    sorted_arr;           // Output: ['a', 'b', 'c', 'd']

Objects

    var example_obj = {
        first_name:'Jim',
        last_name:'Johnson',
        age:88,
        getFullName: function(){
            return this.last_name + ', ' + this.first_name;
        }
    }

    // Two ways of retrieving the values assigned to each property
    example_obj.first_name;     // Output: 'Jim'
    example_obj['last_name'];   // Output: 'Johnson'

    // Updating property values
    example_obj.age = 102;

    // Calling the function assigned to an object's property
    example_obj.getFullName();  // Output: 'Johnson, Jim'
